def some_random_coeffs():
    a = 10 * np.random.rand() - 2.5
    b = 20 * np.random.rand() - 5
    c = 100 * np.random.rand() - 50
    return a, b, c

def some_random_data(a, b, c, N):
    x = np.linspace(-N, N, 2 * N + 1)
    sigma = 10 * (np.random.rand(x.shape[0]) + 1)
    y = np.random.normal(a * x**2 + b * x + c, sigma)
    return x, y, sigma

def random_data_with_outliers(a, b, c, N, M):
    x, y, sigma = some_random_data(a, b, c, N)
    idx = np.random.randint(0, 2 * N, M)
    y[idx] = np.min(y) + np.random.rand() * (np.max(y) - np.min(y))
    return x, y, sigma

Számítsuk ki a legjobban illeszkedő modell esetében a redukált $\chi^2$ értékét!¶

$\chi^2$ kiszmolhat a kvetkezkppen:

$$ \chi^2(\mathbf{a}) = \sum_i \frac{\left[ y(\mathbf{a}, \mathbf{x}_i) - y_i \right]^2}{\sigma_i^2} = \sum_i \frac{\left[ \sum_k a_k f_k(\mathbf{x}_i) - y_i \right]^2}{\sigma_i^2} $$

Ebbl pedig tovbbszmtva a reduklt $\chi^2$ rtke:

$$ \chi^2_\nu = \frac{\chi^2}{\nu}, $$

ahol $\nu = N - M$ a szabadsgi fokok szma, azaz a mrsi adatpontok szma mnusz az illesztend paramterek szma. J illesztsrl akkor beszlhetnk, ha $\chi^2_\nu \approx 1$.

Ez alapjn az elbb kiszmolt paramtereket felhasznlva a redullt $\chi^2$ rtke:

def illesztes(x,y,sigma):
    from numpy.polynomial.polynomial import polyval
    Mat = np.empty((x.shape[0],3))
    for i in range(x.shape[0]):   # tervmtrix feltltse a megadott rtkekkel
        for j in range(3):
            Mat[i,j] = (x[i]**j)/sigma[i]
            
    b = np.empty((x.shape[0],1))    # b vektor helynek lefoglalsa
    for i in range(x.shape[0]):     # b vektor feltltse elemekkel
        b[i] = (y[i])/sigma[i]

    M = np.dot(Mat.T,Mat)     # mtrixmveletek elvgzse :
    N = np.dot(Mat.T,b)
    a = np.linalg.solve(M,N)  # 'a' tartalmazza a chi^2-et minimalizl paramtereket
    z = polyval(x,a,False)    # polinomot ksztnk x adataihoz 
                              # felhasznlva az 'a' minimalizlt rtkeket
    return z
  • fggvny a kilg pontok eltntetshez, mely a kilg pontokat mr nem tartalmaz vektorokkal tr vissza;

def kivagas(x,y,sigma):
    z = illesztes(x,y,sigma)
    d = y-z                     # illesztett rtkek s adatok eltrse
    o = 3*d.std()               # 3*szigma
    uj_y = y[abs(d) < o]        # csak a 3*szigma alatti rtkeket tartalmaz adatsorok 
    uj_x = x[abs(d) < o]
    uj_sigma = sigma[abs(d) < o]
    return uj_x, uj_y, uj_sigma

3. feladat¶

rjuk fggvnyt, mely meghatrozza az illesztett paramterek aszimptotikus hibjt s kovariancijt.

A bevezetett $\mathbf{\alpha} = \mathbf{X}^{T} \mathbf{X}$ szimmetrikus mtrixrl belthat, hogy annak $\mathbf{C} = \mathbf{\alpha}^{-1}$ inverze, az n. kovarianciamtrix, jl jellemzi az illesztett paramterek hibjt s kovariancijt. Az $a_k$ paramter hibja egyszeren $\sigma_{a_k}^2 = C_{kk}$, mg az $a_k$ s $a_l$ paramterek kovariancija a $ \textrm{cov}(a_k, a_l) = C_{kl}$ mtrixelem.

In [21]:
def kov(Mat):    # ahol Mat a tervmtrix
    alfa = np.dot(Mat.T,Mat)   # szimmetrikus mtrix
    C = np.linalg.inv(alfa)    # kovariancia mtrix
    return C
  • Mat a tervmtrix,
  • alfa a szimmetrikus mtrix,
  • C pedig a kovarianciamtrix.
